## Рисунок 3.1 — Физическая схема базы данных системы Carvix

```sql
-- Параметры подключения (общие с веб-приложением)
-- DB_HOST=127.0.0.1
-- DB_PORT=5432
-- DB_NAME=carvix
-- DB_USER=root
-- DB_PASSWORD=130507130507

-- Таблица ролей
CREATE TABLE rol (
    id SERIAL PRIMARY KEY,
    nazvanie VARCHAR(50) NOT NULL
);

-- Данные ролей
INSERT INTO rol (id, nazvanie) VALUES
(1, 'Диспетчер'),
(2, 'Водитель'),
(3, 'Механик'),
(4, 'Администратор');

-- Таблица подразделений
CREATE TABLE podrazdelenie (
    id SERIAL PRIMARY KEY,
    nazvanie VARCHAR(100) NOT NULL
);

-- Таблица сотрудников
CREATE TABLE sotrudnik (
    id SERIAL PRIMARY KEY,
    fio VARCHAR(255) NOT NULL,
    login VARCHAR(50) UNIQUE NOT NULL,
    parol_hash VARCHAR(255) NOT NULL,
    license_number VARCHAR(20),
    phone VARCHAR(20),
    rol_id INTEGER REFERENCES rol(id),
    podrazdelenie_id INTEGER REFERENCES podrazdelenie(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индексы для сотрудников
CREATE INDEX idx_sotrudnik_rol_id ON sotrudnik(rol_id);
CREATE INDEX idx_sotrudnik_podrazdelenie_id ON sotrudnik(podrazdelenie_id);

-- Таблица марок
CREATE TABLE marka (
    id SERIAL PRIMARY KEY,
    nazvanie VARCHAR(100) NOT NULL
);

-- Таблица моделей
CREATE TABLE model (
    id SERIAL PRIMARY KEY,
    nazvanie VARCHAR(100) NOT NULL,
    marka_id INTEGER REFERENCES marka(id)
);

-- Таблица транспортных средств
CREATE TABLE transportnoe_sredstvo (
    id SERIAL PRIMARY KEY,
    gos_nomer VARCHAR(20) UNIQUE NOT NULL,
    invent_nomer VARCHAR(20) UNIQUE NOT NULL,
    vin VARCHAR(17) UNIQUE NOT NULL,
    probeg INTEGER DEFAULT 0,
    model_id INTEGER REFERENCES model(id),
    tekuschee_sostoyanie VARCHAR(50) DEFAULT 'Активно',
    to_expiry DATE,
    insurance_expiry DATE,
    podrazdelenie_id INTEGER REFERENCES podrazdelenie(id),
    assigned_driver_id INTEGER REFERENCES sotrudnik(id),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индексы для ТС
CREATE INDEX idx_transportnoe_sredstvo_model_id ON transportnoe_sredstvo(model_id);
CREATE INDEX idx_transportnoe_sredstvo_podrazdelenie_id ON transportnoe_sredstvo(podrazdelenie_id);
CREATE INDEX idx_transportnoe_sredstvo_assigned_driver_id ON transportnoe_sredstvo(assigned_driver_id);

-- Таблица графика работы
CREATE TABLE work_schedule (
    id SERIAL PRIMARY KEY,
    ts_id INTEGER REFERENCES transportnoe_sredstvo(id),
    driver_id INTEGER REFERENCES sotrudnik(id),
    date DATE NOT NULL,
    shift_start TIME,
    shift_end TIME,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индексы для графика
CREATE INDEX idx_work_schedule_ts_id ON work_schedule(ts_id);
CREATE INDEX idx_work_schedule_driver_id ON work_schedule(driver_id);
CREATE INDEX idx_work_schedule_date ON work_schedule(date);

-- Таблица штрафов
CREATE TABLE fines (
    id SERIAL PRIMARY KEY,
    ts_id INTEGER REFERENCES transportnoe_sredstvo(id),
    date DATE NOT NULL,
    amount DECIMAL(10,2) NOT NULL,
    description TEXT,
    status VARCHAR(50) DEFAULT 'Не оплачен',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индексы для штрафов
CREATE INDEX idx_fines_ts_id ON fines(ts_id);
CREATE INDEX idx_fines_date ON fines(date);
CREATE INDEX idx_fines_status ON fines(status);

-- Таблица ТО и Страховки
CREATE TABLE maintenance (
    id SERIAL PRIMARY KEY,
    ts_id INTEGER REFERENCES transportnoe_sredstvo(id),
    date DATE NOT NULL,
    type VARCHAR(50) NOT NULL,
    description TEXT,
    status VARCHAR(50) DEFAULT 'Запланировано',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индексы для ТО
CREATE INDEX idx_maintenance_ts_id ON maintenance(ts_id);
CREATE INDEX idx_maintenance_date ON maintenance(date);
CREATE INDEX idx_maintenance_type ON maintenance(type);

-- Таблица настроек приложения
CREATE TABLE app_settings (
    id SERIAL PRIMARY KEY,
    key VARCHAR(100) UNIQUE NOT NULL,
    value TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Индекс для настроек
```
